CD8/Ki67 by regression

# Loading the datasets
load("../R/df.Rdata")
# Sourcing scripts
scripts <- list.files(
    path = "~/Dropbox/R-scripts",
    pattern = "*.R$",
    full.names = TRUE
)
for (script in scripts) { source(script) }
# Global options
opts_chunk$set(message = FALSE, warning = FALSE)
# Regression models
df_pairs <- df_pairs %>%
    mutate(
        cd8_tumor_count_pos = ifelse(
            cd8_tumor_count_max > mean(cd8_tumor_count_max, na.rm = TRUE),
            "CD8 positive", "CD8 negative"
        ),
        ki67_tumor_pos = ifelse(
            ki67_tumor_max > mean(ki67_tumor_max, na.rm = TRUE),
            "Ki67 positive", "Ki67 negative"
        ),
        cd8ki67_tumor_count_pos = ifelse(
            cd8ki67_tumor_count_max > mean(cd8ki67_tumor_count_max, na.rm = TRUE),
            "CD8/Ki67 positive", "CD8/Ki67 negative"
        ),
        cd8_stroma_count_pos = ifelse(
            cd8_stroma_count_max > mean(cd8_stroma_count_max, na.rm = TRUE),
            "CD8 positive", "CD8 negative"
        ),
        cd8ki67_stroma_count_pos = ifelse(
            cd8ki67_stroma_count_max > mean(cd8ki67_stroma_count_max, na.rm = TRUE),
            "CD8/ki67 positive", "CD8/Ki67 negative"
        )
    )

predictors <- with(df_pairs, list(
    "CD8 positivity in lymphocytes infiltrating tumor" = cd8_tumor_count_pos,
    "Ki67 positivity in lymphocytes infiltrating tumor" = ki67_tumor_pos,
    "CD8/Ki67 positivity in lymphocytes infiltrating tumor" = cd8ki67_tumor_count_pos,
    "CD8 positivity in lymphocytes infiltrating stroma" = cd8_stroma_count_pos,
    "CD8/Ki67 positivity in lymphocytes infiltrating stroma" = cd8ki67_stroma_count_pos
))

varlabels <- names(predictors)

METHODOLOGY: Odds ratios were estimated using binary logistic regression. CD8 and Ki67 expression were categorized in 2 levels using the meean value as the cutoff point. Thus, any value above the mean was considered as positive.
