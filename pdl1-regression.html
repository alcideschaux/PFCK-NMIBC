PDL1 by regression

# Loading the datasets
load("../R/df.Rdata")
# Loading libraries
library(knitr)
library(dplyr)
# Sourcing scripts
scripts <- list.files(
    path = "~/Dropbox/R-scripts",
    pattern = "*.R$",
    full.names = TRUE
)
for (script in scripts) { source(script) }
# Global options
opts_chunk$set(message = FALSE, warning = FALSE)
# Regression models
df_pairs <- df_pairs %>% 
    mutate(
        pdl1_tumor_pos = ifelse(pdl1_tumor_max > 0, "PDL1 positive", "PDL1 negative"),
        pdl1_tumor_h_pos = ifelse(pdl1_tumor_h_max > 0, "PDL1 positive", "PDL1 negative"),
        pdl1_stroma_pos = ifelse(pdl1_stroma_max > 0, "PDL1 positive", "PDL1 negative")
    )
predictors <- list(
    "Percentage of PDL1 positivity in tumor cells" = df_pairs$pdl1_tumor_pos,
    "PDL1 H-scores in tumor cells" = df_pairs$pdl1_tumor_h_pos,
    "Percentage of PDL1 positivity in stromal cells" = df_pairs$pdl1_stroma_pos
)
varlabels <- names(predictors)

METHODOLOGY: Odds ratios were estimated using binary logistic regression. PDL1 expression was categorized in 2 levels using 0 as the cutoff point. Thus, any >0 value (either percentage or H-score) was considered as positive.
